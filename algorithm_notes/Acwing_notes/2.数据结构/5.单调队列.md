# 单调队列

## 滑动窗口
给定一个大小为 n≤1e6的数组。

有一个大小为 k的滑动窗口，它从数组的最左边移动到最右边。

你只能在窗口中看到 k个数字。

每次滑动窗口向右移动一个位置。

以下是一个例子：

该数组为 [1 3 -1 -3 5 3 6 7]，k为 3

![QQ截图20230312235030](https://raw.githubusercontent.com/Rander7/Note_picture/master/QQ%E6%88%AA%E5%9B%BE20230312235030.png)


你的任务是确定滑动窗口位于每个位置时，窗口中的最大值和最小值。

输入格式

输入包含两行。第一行包含两个整数 n和 k，分别代表数组长度和滑动窗口的长度。第二行有 n个整数，代表数组的具体数值。同行数据之间用空格隔开。

输出格式

输出包含两个，第一行输出，从左至右，每个位置滑动窗口中的最小值。第二行输出，从左至右，每个位置滑动窗口中的最大值。


**方法一：队列存放元素下标**

```
#include<iostream>
using namespace std;

const int maxn=1e6+10;

int a[maxn];
int q[maxn];
int n,k;


int main()
{
	//队列中存放下标 
	int hh=0,tt=-1;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	
	//求最小窗口序列
	for(int i=0;i<n;i++){
		//如果队列不为0，判断队首元素是否超出窗口区间
		if(tt>=hh&&k<i-q[hh]+1){
			hh++;
		} 
		//如果队列不为空且队尾元素大于等于新要入队的元素，则队尾元素永远不可能出头，出队 
		while(tt>=hh&&a[q[tt]]>=a[i]){
			tt--;
		}
		//不管队列是否为空，将新进窗口区间的元素加入队尾
		q[++tt]=i; 
		//超过窗口区间大小的每轮循环输出当前窗口最小值
		if(i+1>=k){
			printf("%d ",a[q[hh]]);
		} 
	}
	printf("\n");
	hh=0,tt=-1;
	//求最大窗口序列
	for(int i=0;i<n;i++){
		//如果队列不为0，判断队首元素是否超出窗口区间
		if(tt>=hh&&k<i-q[hh]+1){
			hh++;
		} 
		//如果队列不为空且队尾元素小于等于新要入队的元素，则队尾元素永远不可能出头，出队 
		while(tt>=hh&&a[q[tt]]<=a[i]){
			tt--;
		}
		//不管队列是否为空，将新进窗口区间的元素加入队尾
		q[++tt]=i; 
		//超过窗口区间大小的每轮循环输出当前窗口最小值
		if(i+1>=k){
			printf("%d ",a[q[hh]]);
		} 
	}
	
	return 0;
}
```


**方式二：队列存放元素值**

```
//单调队列：入队元素值，比较遵从严格大于或小于 
#include<bits/stdc++.h>
using namespace std;

const int N = 1000010;
int a[N];

int main()
{
	int n,k;
	cin>>n>>k;
	deque<int> q;
	for(int i=1;i<=n;i++){//做到从1开始 
		scanf("%d",&a[i]);
	}
	
	//移动窗口最小值序列 
	for(int i=1;i<=n;i++){
		//如果队列不为空 首先循环判断新入队的元素和队尾元素大小
		//如果新来的比当前队尾还要小，这个队尾就可以寄了 
		//注意是要小于队尾 
		while(q.size()&&q.back()>a[i]){
			q.pop_back();
		}
		//如果队列为空或者新来的更大，就入队
		q.push_back(a[i]);
		//判断队首是否还在当前窗口内，如果不在就可以寄
		if(i-k>=1&&a[i-k]==q.front()){
			q.pop_front();
		} 
		//窗口形成，之后每循环一次输出一次队首，这时候队列满足严格单调递增 
		if(i>=k){
			cout<<q.front()<<" ";
		}
	}
	cout<<endl;
	q.clear();

	//移动窗口最大值序列 
	for(int i=1;i<=n;i++){
		while(q.size()&&q.back()<a[i]){
			q.pop_back();
		}
		q.push_back(a[i]);
		if(i-k>=1&&a[i-k]==q.front()){
			q.pop_front();
		}
		if(i>=k){
			cout<<q.front()<<" ";
		}
	} 
	
	return 0;
}
```
