# Dijkstra

## Dijkstra求最短路 I
给定一个 n个点 m条边的有向图，图中可能存在重边和自环，所有边权均为正值。

请你求出 1号点到 n号点的最短距离，如果无法从 1号点走到 n号点，则输出1

输入格式

第一行包含整数 n和 m

接下来 m行每行包含三个整数 x,y,z，表示存在一条从点 x到点 y的有向边，边长为 z

输出格式

输出一个整数，表示 1号点到 n号点的最短距离。

如果路径不存在，则输出-1。

数据范围

1≤n≤500,1≤m≤105,图中涉及边长均不超过10000。

```
#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

const int maxn=510;

int g[maxn][maxn];
bool vis[maxn];
int dist[maxn];//存储当前节点到开始节点的最短路径

int n,m;

int dijkstra()
{
	memset(dist,0x3f,sizeof(dist));
	//这里不能加vis[1]=true，因为他还没有被使用 
	dist[1]=0;
	
	for(int i=0;i<n;i++)//用来循环n次，保证每个与开始节点有边的节点都更新一次 
	{
		
		int t=-1;//存储当前节点
		
        //每次选最短的更新,更新一次就再也不使用他
		for(int j=1;j<=n;j++){
			//如果是当前轮次的节点或者 
			//如果某个节点没有被访问过，并且该节点到开始节点的最短路径小于当前这个轮次的节点 
			if(!vis[j]&&(t==-1||dist[j]<dist[t])){
				t=j;//就用j节点来更新所有相关的路径 
			}
		} 
		vis[t]=true;//用来更新的节点要标记已经使用过 
		
		for(int j=1;j<=n;j++){
			//开始节点到j节点距离如果是到t再到j更小，则更新 
			dist[j]=min(dist[j],dist[t]+g[t][j]);
		}
	}
	
	if(dist[n]==0x3f3f3f3f){
		return -1;
	}
	else{
		return dist[n];
	}

	
}


int main()
{
	scanf("%d%d",&n,&m);
	
	memset(g,0x3f,sizeof(g));
	
	while(m--)
	{
		int x,y,z;
		scanf("%d%d%d",&x,&y,&z);
		g[x][y]=min(g[x][y],z);
	}
	
	printf("%d\n",dijkstra());
	
	
	
	return 0;
} 
```



## Dijkstra求最短路 II(堆优化)
*点多边少，用邻接表*


给定一个 n个点 m条边的有向图，图中可能存在重边和自环，所有边权均为非负值。

请你求出 1号点到 n号点的最短距离，如果无法从 1号点走到 n号点，则输出1
。

输入格式

第一行包含整数 n和 m


接下来 m行每行包含三个整数 x,y,z，表示存在一条从点 x到点 y的有向边，边长为 z

输出格式

输出一个整数，表示 1号点到 n号点的最短距离。

如果路径不存在，则输出-1

数据范围

1≤n,m≤1.5×10^5,图中涉及边长均不小于 0，且不超过 10000

如果最短路存在，则最短路的长度不超过 10^9

```
#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=2e5+10;

typedef pair<int,int> PII;

int dist[maxn],e[maxn],ne[maxn],h[maxn];
bool vis[maxn];
int w[maxn];

int n,m,idx;

void add(int x,int y,int c)
{
	//不用考虑重边，重复添加没关系，使用小根堆
	//大的边在下面，去了这个节点之后就标记了，不会再取到
	w[idx]=c;
	e[idx]=y; 
	ne[idx]=h[x];
	h[x]=idx++;
}

int dijkstra()
{
	memset(dist,0x3f,sizeof(dist));
	priority_queue<PII,vector<PII>,greater<PII>> heap;
	
	dist[1]=0;
	heap.push({0,1});//前面是距离后面是序号,保证顺序，按距离大小排序插入 
	
	while(heap.size())
	{
		auto temp=heap.top();
		heap.pop();
		
		int ver=temp.second,distance=temp.first;
		
		if(vis[ver]){
			continue;
		}
		vis[ver]=true;
		
		for(int i=h[ver];i!=-1;i=ne[i]){
			int j=e[i];
			if(dist[j]>distance+w[i]){
				dist[j]=distance+w[i];
				heap.push({dist[j],j});
			}
		}
	}
	
	if(dist[n]==0x3f3f3f3f){
		return -1;
	}
	else{
		return dist[n];
	}

}



int main()
{
	memset(h,-1,sizeof(h));
	scanf("%d%d",&n,&m);
	while(m--)
	{
		int x,y,c;
		scanf("%d%d%d",&x,&y,&c);
		add(x,y,c);
	}
	
	printf("%d\n",dijkstra());
	
	
	
	return 0;
}
```
