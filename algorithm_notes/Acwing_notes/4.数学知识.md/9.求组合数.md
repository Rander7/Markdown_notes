# 求组合数

给定 n组询问，每组询问给定两个整数 a，b，请你输出 Cba mod(10^9+7)的值。

## 求组合数 I
当ab比较小时候,直接开二维数组，利用递推公式
$$ C_{a}^{b}=C_{a-1}^{b-1}+C_{a-1}^{b}$$

1≤n≤10000,1≤b≤a≤2000

```
#include<iostream>
using namespace std;

const int mod1=1e9+7;
const int maxn=2010;

int c[maxn][maxn];
int n;

void init()
{
	for(int i=0;i<=2000;i++){
		for(int j=0;j<=i;j++){
			if(!j){
				c[i][j]=1;
			}
            //i=0的时候，j也只有等于0的一个情况，根本执行不到c[i - 1][j] + c[i - 1][j - 1]这个情况
            //所以不会越界
			else{
				c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod1;				
			}
		}
	}		
}


int main()
{
	scanf("%d",&n);
	init();
	
	while(n--)
	{
		int a,b;
		scanf("%d%d",&a,&b);
		cout<<c[a][b]<<endl;
	}
	
	return 0;
}
```

##  求组合数 II

当ab比较大，无法直接开二维数组,

直接利用公式原本定义
$$C_{a}^{b}=a!/(b!*(a-b)!)$$

使用乘法逆元求解

1≤b≤a≤10^5

```
#include<iostream>
#include<algorithm>

using namespace std;

const int maxn=100010,mod1=1e9+7;

int fact[maxn],infact[maxn];//fact表示阶乘，inface表示阶乘的逆元

int quick_m(int a,int k,int p)
{
	int res=1;
	while(k)
	{
		if(k&1){
			res=(long long)res*a%p;
		}
		a=(long long)a*a%p;
		k>>=1;
	}
	return res;
}


int main()
{
	fact[0]=infact[0]=1;
	for(int i=1;i<maxn;i++){
		fact[i]=(long long)fact[i-1]*i%mod1;
		infact[i]=(long long)infact[i-1]*quick_m(i,mod1-2,mod1)%mod1; 
	}
	
	int n;
	scanf("%d",&n);
	while(n--)
	{
		int a,b;
		scanf("%d%d",&a,&b);
		printf("%d\n",(long long)fact[a]*infact[a-b]%mod1*infact[b]%mod1);
	
	}
	
	return 0;	
} 

```

### 求组合数 III

当ab非常大，是使用卢卡斯定理递归分布求解

1≤n≤20,1≤b≤a≤10^18,1≤p≤10^5

```
#include<iostream>

using namespace std;

int p;

long long quick_m(long long a,long long s)
{
	long long res=1;
	while(s)
	{
		if(s&1){
			res=res*a%p;
		}
		a=(a*a)%p;
		s>>=1;
	}
	return res;
	
}

//直接法求组合数
//分子从a乘到a-b+1，分子是b的阶乘 ，总共循环次数为b
//j表示分子，利用乘法逆元让res每次乘上b某个阶乘项的逆元 
long long c(long long a,long long b)
{
	long long res=1;
	for(int i=1,j=a;i<=b;++i,--j){
		res=res*j%p;
		res=res*quick_m(i,p-2)%p;
	}
	return res%p;
}

long long lucas(long long a,long long b)
{
	if(a<p&&b<p){
		return c(a,b);
	}
	else{
		return c(a%p,b%p)*lucas(a/p,b/p)%p;
	}
}


int main()
{
	int n;
	scanf("%d",&n);
	while(n--)
	{
		long long a,b;
		scanf("%lld%lld%d",&a,&b,&p);
		printf("%lld\n",lucas(a,b));	
	}	
	
	return 0;
}
```


## 求组合数 IV

结果很大要使用高精度计算

每个数能分解成素数的乘积，对于公式

$$C_{a}^{b}=a!/(b!*(a-b)!)$$
可以将分子分母每一项分解成素数的乘积式,在乘积式中把素数的次幂相减

1≤b≤a≤5000

```
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

const int maxn=5010;
int primes[maxn],cnt=0;
bool vis[maxn];
int sum[maxn];

void get_primes(int n)
{
	for(int i=2;i<=n;i++){
		if(!vis[i]){
			primes[cnt++]=i;
		}
		for(int j=0;primes[j]<=n/i;j++){
			vis[primes[j]*i]=true;
			if(i%primes[j]==0){
				break;
			}
		}
	}
}

int get(int n,int p)
{
	int res=0;
	while(n)
	{
		res+=n/p;
		n/=p;
	}
	return res;
}

vector<int> mul(vector<int> a,int b)
{
	vector<int> c;
	int t=0;
	for(int i=0;i<a.size();i++){
		t+=a[i]*b;
		c.push_back(t%10);
		t/=10;
	}
	while(t)
	{
		c.push_back(t%10);
		t/=10;
	}
	while(c.size()>1&&c.back()==0){
		c.pop_back();
	}
	return c;
}


int main()
{
	int a,b;
	scanf("%d%d",&a,&b);
	get_primes(a);
	
	for(int i=0;i<cnt;i++){
		int p=primes[i];
		sum[i]=get(a,p)-get(a-b,p)-get(b,p);
	}
	
	vector<int> res;
	res.push_back(1);
	
	for(int i=0;i<cnt;i++){
		for(int j=0;j<sum[i];j++){
			res=mul(res,primes[i]);
		}
	}
	
	for(int i=res.size()-1;i>=0;i--){
		printf("%d",res[i]);
	}
	printf("\n");
	
	return 0;
}
```
