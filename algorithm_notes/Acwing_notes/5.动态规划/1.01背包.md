# 01背包

## 每个物品最多只拿一次
https://www.acwing.com/problem/content/2/

1. 朴素写法
```
#include<iostream>
using namespace std;

const int maxn=1010;

int f[maxn][maxn];
//v，w分别代表体积和价值 
int v[maxn],w[maxn];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&v[i],&w[i]);
	}
	
	//初始化没有放物品时候背包当前容量
	//因为采用全局变量所以省略初始化 
    //	for(int i=0;i<m;i++){
    //		f[0][i]=0;
    //	}
	 
	for(int i=1;i<=n;i++){
		for(int j=0;j<=m;j++){
			//如果当前背包装不下当前物品，价值等于前i-1个物品 
			f[i][j]=f[i-1][j];
			//如果装的下，判断是否继续装第i个物品 
			if(j>=v[i]){
				//后面式子代表装第i个前的背包状态+第i个物品 
				f[i][j]=max(f[i][j],f[i-1][j-v[i]]+w[i]);
			}
		}
	}
	printf("%d",f[n][m]);
	
	
	return 0;
}

```


2. 优化成一维写法

```
#include<iostream>
using namespace std;

const int maxn=1010;

int f[maxn];
int v[maxn],w[maxn];

//将状态是否更新分成两种情况：
//1. 对于第i个物品不加入背包f[i][j]=f[i-1][j]
//2. 准备将第i个物品加入背包但是还要判断是否合理
//f[i][j]=max(f[i-1][j],f[i-1][j-v[i]]+w[i])
//假设加入了i物品，将其先去除，判断去除后状态加上i物品后
//和之前没有考虑加入i物品的状态比较大小 

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	
	for(int i=1;i<=n;i++){
		scanf("%d%d",&v[i],&w[i]);
	}

	//总共i个物品，循环i层判断是否要将第i个物品加入背包 
	for(int i=1;i<=n;i++){
		//从大到小更新背包状态
		//j-v[i]<j,如果j从小到大递增的话，
		//j-v[i]一定是第i层(当前层)的背包状态
		//我们需要f[i-1][j-v[i]]也就是上一层的背包状态
		//所以如果从大到小递减，j-v[i]<j,当前层的f[j-v[i]]
		//还没有更新到，说明一定用的是上一层的f[j-v[i]] 
		for(int j=m;j>=v[i];j--){
			//当前层背包状态
			//f[i][j]=max(f[i-1][j],f[i-1][j-v[i]+w[i]) 
			//去掉第一维后，存在于f[j]和f[j-v[i]]中的值
			//是上一次外循环(i-1轮)更新的数值 
			f[j]=max(f[j],f[j-v[i]]+w[i]);
		}
	}
	
	printf("%d\n",f[m]);
	
	
	return 0;
}
```
