# 完全背包

## 每个物品可以拿一次或无限次


1. 朴素写法
```
#include<iostream>
using namespace std;

const int maxn=1010;

int f[maxn][maxn];
int v[maxn],w[maxn];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&v[i],&w[i]);
	} 
	

//当前背包状态有3种情况 
//1. 第一种加入k个第i个物品
//f[i][j]=f[i-1][j-k*v[i]]+k*w[i]
//2. 不选，最优解为前i-1物品的所有选择中
//的最大价值，即 f[i][j]=f[i-1][j] 
//3. 还有一种情况，因为是三重循环，在某次k循环中
//f[i][j]已经被更新，这次k循环只不过是k的大小不同

	for(int i=1;i<=n;i++){
		for(int j=0;j<=m;j++){
			for(int k=0;k*v[i]<=j;k++){
				f[i][j]=max(f[i][j],f[i-1][j-k*v[i]]+k*w[i]);
			}
		}
	}
	
	printf("%d\n",f[n][m]);
	
	return 0;
}
```

2. 优化写法
```
#include<iostream>
using namespace std;

const int maxn=1010;

int f[maxn][maxn];
int v[maxn],w[maxn];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&v[i],&w[i]);
	} 
	

//转移方程 
//dp[i][j]=max(dp[i-1][j],dp[i-1][j-v]+w,dp[i-1][j-2v]+2w,..,dp[i-1][j-kv]+kw)
//将左式的j替换成 j-v
//dp[i][j-v]=max(dp[i-1][j-v],dp[i-1][j-2v]+w,..,dp[i-1][j-kv]+(k-1)w)
//两者之间的差异仅为一个 w
//dp[i][j]=max(dp[i-1][j],dp[i][j-v]+w)

//这个j-v没有实际含义，单纯是数学递推，只需要记住是第i层
//01背包这里是第i-1层 

	for(int i=1;i<=n;i++){
		for(int j=0;j<=m;j++){
			f[i][j]=f[i-1][j];
			if(j>=v[i]){
				f[i][j]=max(f[i-1][j],f[i][j-v[i]]+w[i]);				
			}
		}
	}
	
	printf("%d\n",f[n][m]);
	
	return 0;
}
```

3. 最佳优化写法(将数组降至一维)
```
#include<iostream>
using namespace std;

const int maxn=1010;

int f[maxn];
int v[maxn],w[maxn];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&v[i],&w[i]);
	} 
	
	for(int i=1;i<=n;i++){
		//这里j从小到大递增，j>j-v[i]所以算f[j]
		//时候f[j-v[i]]已经没计算过了，所以是第i层的 
		for(int j=v[i];j<=m;j++){
			f[j]=max(f[j],f[j-v[i]]+w[i]);				
		}
	}
	
	printf("%d\n",f[m]);
	
	return 0;
}
```
