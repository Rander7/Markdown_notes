# 数字三角形

```

#include<iostream>
using namespace std;

const int maxn=510,INF=1e9;
int n;
int ans=-INF;

int f[maxn][maxn];

int main()
{
	scanf("%d",&n);
    //记得多初始化一位，因为最右边的数递推时候要使用超出三角形范围的数
	for(int i=0;i<=n;i++){
		for(int j=0;j<=i+1;j++){
			f[i][j]=-INF;
		}
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=i;j++){
			scanf("%d",&f[i][j]);
		}
	}
	
	for(int i=2;i<=n;i++){
		for(int j=1;j<=i;j++){
			f[i][j]=f[i][j]+max(f[i-1][j],f[i-1][j-1]);
		}
	}
	
	for(int i=1;i<=n;i++){
		if(f[n][i]>ans){
			ans=f[n][i];
		}
	}
	printf("%d\n",ans);
	
	
	return 0;
}
```
