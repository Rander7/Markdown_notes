# 高精度问题

写在前面：高精度算法之间有一套固定模板，其中加乘相似，减除有所变形，再就是要注意去除首位冗余的0问题

高精度算法考虑进位借位时候用t加上或者赋予总和，再取余或者其他操作

## 高精度加法

```
#include<iostream>
#include<vector>

using namespace std;

//注意细节传入vector用引用
vector<int> add(vector<int> &A,vector<int> &B){
    vector<int> C;
    int t = 0;//初始化进位t
    //高精度加法一直加结束条件是两个数组都到界限
    for (int i = 0; i < A.size() || i < B.size(); i ++ ){
        //先全部加到t上，再把个位push_back，进位/10
        //两个数组要是还有位就继续加，没有不单独开条件加另一个直接不管就是
        if(i < A.size()) t += A[i];
        if(i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }
    //考虑最终位进位问题
    if (t == 1) C.push_back(1);
    return C;
}


//输入有一套固定模板：利用string输入
int main(){
    string a,b;
    cin>>a>>b;
    vector<int> A,B;
    //将AB中数字逆序存储
    for(int i = a.size()-1;i >= 0;i--){
        A.push_back(a[i] - '0');
    }
    for(int i = b.size()-1;i >= 0;i--){
        B.push_back(b[i] - '0');
    }
    //将运算结果初始化C，再逆序输出C
    vector<int> C = add(A, B);
    for (int i = C.size() - 1; i >= 0; i --) cout<<C[i];
    return 0;
}

```



## 高精度减法

减法要考虑的问题比加法多一步，每次一定用大数减小数，所以加一个判断大小函数

```
#include<iostream>
#include<vector>

using namespace std;

vector<int> v1,v2;


int vcmp(vector<int> &v1,vector<int> &v2)
{
    //判断两个数是否大小相等，首先判断位数是不是相等
	if(v1.size()!=v2.size()){
		return v1.size()>v2.size();
	}
    //位数相等依次比较相同位置上数是否相等，找到第一个不相等的比较
	for(int i=v1.size()-1;i>=0;i--){
		if(v1[i]!=v2[i]){
			return v1[i]>v2[i];
		} 
	}
    //如果两个数相等，我们这里返回真值，因为有负数输出加负号问题，0是不带符号的，所以归类到A>B一类
	return 1;
}

vector<int> vsub(vector<int>& v1,vector<int>& v2)
{
	vector<int> res;
	int t=0;
	int i;
    //因为前面做过处理，所以v1一定大于v2
	for(i=0;i<v1.size();i++){
        //每次用t来实行包含所有的操作
		t=v1[i]-t;//减去上一个数位的借位项
		if(i<v2.size()) t-=v2[i];//如果v2还不为空，就继续减v2
		res.push_back((t+10)%10);//保证是正数把其加10再%10
        //如果t<0就是要皆为了，下一位计算时候要减t
		if(t<0){
			t=1;
		}
		else{
			t=0;
		}
	}
    //去除前面冗余的0
    //如果只有一个0就不要去了，不然输出不了0
	while(res.back()==0&&res.size()>1){	
		res.pop_back();
	}
	return res;
}


int main()
{
	vector<int>res;
	string s1,s2;
	cin>>s1>>s2;
	for(int i=s1.size()-1;i>=0;i--){
		v1.push_back(s1[i]-'0');
	}
	for(int i=s2.size()-1;i>=0;i--){
		v2.push_back(s2[i]-'0');
	}
	
	if(vcmp(v1,v2)){
		res=vsub(v1,v2);
	}
	else{
		res=vsub(v2,v1);
		cout<<"-";
	}
	for(int i=res.size()-1;i>=0;i--){
		cout<<res[i];
	}

	return 0;
}
```



## 高精度乘法

*1. 高精度×低精度*

```
#include <iostream>
#include <vector>

using namespace std;

vector <int> mul(vector <int> & A, int b) {
    vector <int> C;

    int t = 0;
    //比较特殊的在于每一个位乘的是整个b
    for (int i = 0; i < A.size(); i ++) {
        t += A[i] * b;       // t + A[i] * b = 7218
        C.push_back(t % 10); // 只取个位 8
        t /= 10;             // 721 看作 进位
    }

    while (t) {            // 处理最后剩余的 t
        C.push_back(t % 10);
        t /= 10;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();

    return C;
}

int main() {
    string a;
    int b;
    cin >> a >> b;

    vector <int> A;
    for (int i = a.size() - 1; i >= 0; i --) A.push_back(a[i] - '0');

    auto C = mul(A, b);

    for (int i = C.size() - 1; i >= 0; i --) {
        cout << C[i];
    }

    return 0;
}
```

*2. 高精度×高精度*

```
#include <iostream>
#include <vector>

using namespace std;

vector<int> mul(vector<int> &A, vector<int> &B) {
    vector<int> C(A.size() + B.size() + 7, 0); // 初始化为 0，C的size可以大一点

    for (int i = 0; i < A.size(); i++)
        for (int j = 0; j < B.size(); j++)
            C[i + j] += A[i] * B[j];

    int t = 0;
    for (int i = 0; i < C.size(); i++) { // i = C.size() - 1时 t 一定小于 10
        t += C[i];
        C[i] = t % 10;
        t /= 10;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back(); // 必须要去前导 0，因为最高位很可能是 0
    return C;
}

int main() {
    string a, b;
    cin >> a >> b; // a = "1222323", b = "2323423423"

    vector<int> A, B;
    for (int i = a.size() - 1; i >= 0; i--)
        A.push_back(a[i] - '0');
    for (int i = b.size() - 1; i >= 0; i--)
        B.push_back(b[i] - '0');

    auto C = mul(A, B);

    for (int i = C.size() - 1; i >= 0; i--)
        cout << C[i];

    return 0;
}
```



## 高精度除法

和其他不同在于他从最高位开始处理,存的是顺序，处理也是顺序，但是去0要反转pop_back

```
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int r=0;
vector<int> div(vector<int> &A,int B,int &r){//r传入r的地址，便于直接对余数r进行修改
    vector<int> C;
    for(int i=0;i<A.size();i++){//对A从最高位开始处理
        r=r*10+A[i];//将上次的余数*10在加上当前位的数字，便是该位需要除的被除数
        C.push_back(r/B);//所得即为商在这一位的数字
        r=r%B;
    }
    //由于在除法运算中，高位到低位运算，因此C的前导零都在vector的前面而不是尾部，vector只有删除最后一个数字pop_back是常数复杂度，而对于删除第一位没有相应的库函数可以使用，而且删除第一位，其余位也要前移，
    //因此我们将C翻转，这样0就位于数组尾部，可以使用pop函数删除前导0
    reverse(C.begin(),C.end());
    while(C.size()>1&&C.back()==0) C.pop_back();
    return C;
}
int main(){
    string a;
    int B,r=0; //代表余数
    cin>>a>>B;
    vector<int> A;
    for(int i=0;i<a.size();i++) A.push_back(a[i]-'0');//注意这次的A是由高为传输至低位，由于在除法的手算过程中，发现从高位进行处理
    //for(int i=0;i<A.size();i++) cout<<A[i];
    //cout<<B;
    auto C = div(A,B,r);
    for(int i=C.size()-1;i>=0;i--) cout<<C[i];//将C从最高位传给最低位
    cout<<endl<<r;//输出余数
    cout<<endl;
    return 0;
}
```
